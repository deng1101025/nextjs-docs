## 简单深拷贝

```js
// 深拷贝
deepCopy(obj) {
    if(typeof obj !== 'object') return obj

    let ret = Array.isArray(obj) ? [] : {}
    for (const key in obj) {
        if (Object.hasOwnProperty.call(obj, key)) {
            const value = obj[key];
            if(typeof value !== 'object') {
                ret[key] = value
            } else {
                ret[key] = this.deepCopy(value)
            }
        }
    }

    return ret
}
```
